{{/* Renders Solo CLI command output fetched from the latest release artifact at build time. Parameters: ref – key in solo-command-output.json (e.g. "solo-consensus-node-start") lang – syntax-highlight language (default "text") Usage: {{< solo-output ref="solo-consensus-node-start" lang="bash" >}} */}} {{- $ref := .Get "ref" -}} {{- $lang := .Get "lang" | default "text" -}} {{- $npm := dict -}} {{- with resources.GetRemote "https://registry.npmjs.org/@hiero-ledger%2Fsolo" -}} {{- $npm = .Content | transform.Unmarshal -}} {{- else -}} {{- errorf "solo-output: could not fetch npm metadata for @hiero-ledger/solo" -}} {{- end -}} {{- $version := index $npm "dist-tags" "latest" -}} {{- $tag := printf "v%s" $version -}} {{- $url := printf "https://github.com/hiero-ledger/solo/releases/download/%s/solo-command-output.json" $tag -}} {{- $data := dict -}} {{- with resources.GetRemote $url -}} {{- $data = .Content | transform.Unmarshal -}} {{- else -}} {{- errorf "solo-output: could not fetch command-output artifact for tag %s" $tag -}} {{- end -}} {{- $encoded := index $data $ref -}} {{- if not $encoded -}} {{- errorf "solo-output: ref %q not found in artifact for tag %s" $ref $tag -}} {{- end -}} {{- highlight ($encoded | base64Decode) $lang "" -}}
